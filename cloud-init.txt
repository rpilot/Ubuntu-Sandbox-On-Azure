#cloud-config
package_update: true

packages:
  - xfce4
  - xfce4-goodies
  - xrdp
  - git
  - python3
  - python3-pip
  - curl
  - wget
  - gpg

runcmd:
  # Enable RDP
  - systemctl enable xrdp
  - systemctl start xrdp

  # XFCE session setup
  - echo "startxfce4" > /home/azureuser/.xsession
  - chown azureuser:azureuser /home/azureuser/.xsession

  # Install VS Code (official Microsoft repo)
  - wget -qO- https://packages.microsoft.com/keys/microsoft.asc | gpg --dearmor > packages.microsoft.gpg
  - install -o root -g root -m 644 packages.microsoft.gpg /etc/apt/trusted.gpg.d/
  - sh -c 'echo "deb [arch=amd64 signed-by=/etc/apt/trusted.gpg.d/packages.microsoft.gpg] https://packages.microsoft.com/repos/code stable main" > /etc/apt/sources.list.d/vscode.list'
  - apt-get update
  - apt-get install -y code

  # Create workspace
  - mkdir -p /home/azureuser/workspace
  - chown -R azureuser:azureuser /home/azureuser/workspace
